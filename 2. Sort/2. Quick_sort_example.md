# Quick sort

- HackerRank(https://www.hackerrank.com/challenges/quicksort1/problem) 
- inputì„ quick_sort


```python
num = int(input())
arr = list(map(int,input().split()))
result = list()

def quick_sort(data):
    
    if len(data) <= 1:
        return data
    
    pivot = data[0]
    left, right = list(), list()
    
    for i in range(1,len(data)):
        if data[i] <= pivot:
            left.append(data[i])
        else:
            right.append(data[i])
    
    quick_sort(left)
    quick_sort(right)
    
    result = quick_sort(left) + [pivot] + quick_sort(right)
    return result

result = quick_sort(arr)
for i in range(len(result)):
    print(result[i],end=' ')
```
